\begin{problem}
  Construct NFAs for the languages generated by each of the following regular expressions. 
  \begin{enumalph}
    \item $10\; \cup \; (0 \; \cup \; 11)0^*1$
    \begin{Answer}
      % Draw the NFA
        \begin{figure}[H]
          \centering
          \begin{tikzpicture}
            \node[state, initial] (q0) at (0, 0) {$q_0$};
            \node[state, right of=q0] (q1) {$q_1$};
            \node[state, below of=q0] (q2) {$q_2$};
            \node[state, right of=q2, accepting] (q3) {$q_3$};
            \node[state, right of=q3] (q4) {$q_4$};
            \draw (q0) edge[above] node {$0$} (q1)
                  (q0) edge[left] node {$1$} (q2)
                  (q1) edge[loop right, right] node {$0$} (q1)
                  (q1) edge[right] node {$1$} (q3)
                  (q2) edge[below] node {$0$} (q3)
                  (q2) edge[above] node {$1$} (q1)
                  (q3) edge[below] node {$0, 1$} (q4)
                  (q4) edge[loop right, right] node {$0, 1$} (q4);
          \end{tikzpicture}
          \caption{NFA to accept $10\; \cup \; (0 \; \cup \; 11)0^*1$}
          \label{fig:1a}
        \end{figure}
      
    \end{Answer}
    \newpage
    \item $(0\; \cup\; 1)((0\; \cup\; 1) (0\; \cup \; 1))^* \cup
      ((0\; \cup \; 1)(0\; \cup \; 1)(0\; \cup \; 1))^*$
    \begin{Answer}
      Observation: the regular expression matches all strings
      of length $n$ such that either $n \pmod 2 \equiv 1$
      or $n \pmod 3 \equiv 0$.

      % Draw the NFA
      \begin{figure}[H]
        \centering
        \begin{tikzpicture}
          \crim{
            \node[state, accepting](q0) at (0, 5) {$q_0$};
            \node[state, right of=q0] (q1) {$q_1$};
            \node[state, right of=q1] (q2) {$q_2$};
            \draw (q0) edge[below] node {$0, 1$} (q1)
                  (q1) edge[below] node {$0, 1$} (q2)
                  (q2) edge[above, bend right] node {$0, 1$} (q0);
          }
          \zaff{
            \node[state, below of=q0](q-even) {$q_{even}$};
            \node[state, below of=q2, accepting] (q-odd) {$q_{odd}$};
            \draw (q-even) edge[above, bend left] node {$0, 1$} (q-odd)
                  (q-odd) edge[above, bend left] node {$0, 1$} (q-even);
          }
          \node[state, initial, left of=q0] (start) {$start$};
          \draw (start) edge[above] node {$\eps$} (q0);
          \draw (start) edge[below] node {$\eps$} (q-even);
        \end{tikzpicture}
        \caption{NFA to accept $(0\; \cup\; 1)((0\; \cup\; 1) (0\; \cup \; 1))^* \cup
          ((0\; \cup \; 1)(0\; \cup \; 1)(0\; \cup \; 1))^*$}
        \label{fig:1b}
      \end{figure}

    \end{Answer}
  \end{enumalph}
\end{problem}
